// stores/dashboard.ts
import { defineStore } from 'pinia'

export const useDashboardStore = defineStore('dashboard', {
  state: () => ({
    coreMetrics: {} as CoreMetrics,
    revenueData: [] as RevenueData[],
    orderDistribution: [] as OrderDistribution[],
    topDishes: [] as TopDish[],
    customerFlow: [] as CustomerFlow[],
    recentOrders: [] as Order[],
    loading: false,
    error: null as string | null
  }),

  getters: {
    totalRevenue: (state) => state.coreMetrics.totalRevenue || 0,
    totalOrders: (state) => state.coreMetrics.totalOrders || 0,
    pendingOrdersCount: (state) => {
      return state.recentOrders.filter(o => o.status === 'pending').length
    }
  },

  actions: {
    async loadDashboardData(params: any) {
      this.loading = true
      try {
        const response = await dashboardApi.getDashboardData(params)
        if (response.success && response.data) {
          this.coreMetrics = response.data.coreMetrics || {}
          this.revenueData = response.data.revenueData || []
          this.orderDistribution = response.data.orderDistribution || []
          this.topDishes = response.data.topDishes || []
          this.customerFlow = response.data.customerFlow || []
          this.recentOrders = response.data.recentOrders || []
        }
        return response
      } catch (error: any) {
        this.error = error.message || 'Failed to load dashboard data'
        throw error
      } finally {
        this.loading = false
      }
    },

    addOrder(order: Order) {
      this.recentOrders.unshift(order)
      if (this.recentOrders.length > 10) {
        this.recentOrders.pop()
      }
    },

    updateOrderStatus(orderId: string, status: string) {
      const index = this.recentOrders.findIndex(order => order.id === orderId)
      if (index !== -1) {
        this.recentOrders[index].status = status
      }
    }
  }
})
